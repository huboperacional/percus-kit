`libs/node/percus-auth/src/tenant/index.ts`:

```typescript
import type { TenantConfig, TenantFetchOptions } from "./types";

const DEFAULT_TTL_MS = 5 * 60 * 1000;
const CACHE_PREFIX = "percus-auth:tenant:";
const memCache = new Map<string, { value: TenantConfig | null; expiresAt: number }>();

export function _clearTenantCache(): void {
  memCache.clear();
}

export async function getTenantConfig(
  opts: TenantFetchOptions
): Promise<TenantConfig | null> {
  const { authServiceUrl, origin, signal, cacheTtlMs = DEFAULT_TTL_MS } = opts;
  const cacheKey = `${authServiceUrl}|${origin}`;

  const memHit = memCache.get(cacheKey);
  if (memHit && memHit.expiresAt > Date.now()) return memHit.value;

  if (typeof sessionStorage !== "undefined") {
    const raw = sessionStorage.getItem(CACHE_PREFIX + cacheKey);
    if (raw) {
      try {
        const parsed = JSON.parse(raw) as { value: TenantConfig | null; expiresAt: number };
        if (parsed.expiresAt > Date.now()) {
          memCache.set(cacheKey, parsed);
          return parsed.value;
        }
      } catch { /* corrupt entry, ignora */ }
    }
  }

  const url = new URL("/tenants/by-origin", authServiceUrl);
  url.searchParams.set("origin", origin);
  const response = await fetch(url.toString(), { signal });

  if (response.status === 404) {
    const entry = { value: null, expiresAt: Date.now() + cacheTtlMs };
    memCache.set(cacheKey, entry);
    if (typeof sessionStorage !== "undefined") {
      sessionStorage.setItem(CACHE_PREFIX + cacheKey, JSON.stringify(entry));
    }
    return null;
  }
  if (response.status === 429) {
    throw new Error("tenant config rate limited (HTTP 429)");
  }
  if (!response.ok) {
    throw new Error(`tenant config fetch failed (HTTP ${response.status})`);
  }

  const config = (await response.json()) as TenantConfig;
  const entry = { value: config, expiresAt: Date.now() + cacheTtlMs };
  memCache.set(cacheKey, entry);
  if (typeof sessionStorage !== "undefined") {
    sessionStorage.setItem(CACHE_PREFIX + cacheKey, JSON.stringify(entry));
  }
  return config;
}
```

```typescript
import * as React from "react";
import type { TenantConfig } from "./types";
import { getTenantConfig } from "./index";

export type TenantStatus = "loading" | "ready" | "error";

export interface TenantContextValue {
  status: TenantStatus;
  config: TenantConfig | null;
  error: Error | null;
  refresh: () => void;
}

const TenantContext = React.createContext<TenantContextValue | undefined>(undefined);

export interface TenantProviderProps {
  authServiceUrl: string;
  origin: string;
  fallback?: TenantConfig;
  children: React.ReactNode;
}

export function TenantProvider({
  authServiceUrl,
  origin,
  fallback,
  children,
}: TenantProviderProps) {
  const [status, setStatus] = React.useState<TenantStatus>("loading");
  const [config, setConfig] = React.useState<TenantConfig | null>(null);
  const [error, setError] = React.useState<Error | null>(null);
  const [tick, setTick] = React.useState(0);

  React.useEffect(() => {
    let cancelled = false;
    setStatus("loading");
    setError(null);
    getTenantConfig({ authServiceUrl, origin })
      .then(c => {
        if (cancelled) return;
        setConfig(c);
        setStatus("ready");
      })
      .catch(err => {
        if (cancelled) return;
        setError(err instanceof Error ? err : new Error(String(err)));
        setConfig(fallback ?? null);
        setStatus("error");
      });
    return () => { cancelled = true; };
  }, [authServiceUrl, origin, tick, fallback]);

  const value = React.useMemo<TenantContextValue>(
    () => ({ status, config, error, refresh: () => setTick(t => t + 1) }),
    [status, config, error]
  );

  return <TenantContext.Provider value={value}>{children}</TenantContext.Provider>;
}

export function useTenant(): TenantContextValue {
  const ctx = React.useContext(TenantContext);
  if (!ctx) {
    throw new Error("useTenant must be used within a TenantProvider");
  }
  return ctx;
}
```

```python
"""Tenant config fetch para consumers Python (Sprint v6.8 Frente D)."""
from __future__ import annotations

from dataclasses import dataclass
from typing import Any

import httpx


@dataclass(frozen=True)
class TenantConfig:
    audience: str
    product_name: str
    logo_url: str | None = None
    logo_url_dark: str | None = None
    palette: dict[str, Any] | None = None
    copy: dict[str, Any] | None = None
    support_contact_url: str | None = None

    @classmethod
    def from_dict(cls, data: dict[str, Any]) -> "TenantConfig":
        return cls(
            audience=data["audience"],
            product_name=data["product_name"],
            logo_url=data.get("logo_url"),
            logo_url_dark=data.get("logo_url_dark"),
            palette=data.get("palette"),
            copy=data.get("copy"),
            support_contact_url=data.get("support_contact_url"),
        )


async def fetch_tenant_config(
    *,
    auth_service_url: str,
    origin: str,
    timeout: float = 5.0,
) -> TenantConfig | None:
    """Busca config de tenant pelo origin. Retorna None se 404. Lança em outros erros."""
    async with httpx.AsyncClient(base_url=auth_service_url, timeout=timeout) as client:
        response = await client.get("/tenants/by-origin", params={"origin": origin})
        if response.status_code == 404:
            return None
        response.raise_for_status()
        return TenantConfig.from_dict(response.json())
```
