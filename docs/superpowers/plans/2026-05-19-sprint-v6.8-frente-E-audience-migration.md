```python
"""rename legacy underscore audiences to kebab-case + populate alias_slugs

Revision ID: NNNN_audience_rename_underscore
Revises: NNNN_audience_branding
Create Date: 2026-05-MM

R7.5 (v6.8.0): kebab-case obrigatório. Audiences com underscore migradas.
alias_slugs preserva o slug antigo por 7 dias pra tolerância cross-repo.
"""
from alembic import op
import sqlalchemy as sa

revision = "NNNN_audience_rename_underscore"
down_revision = "NNNN_audience_branding"  # ajustar pro ID real da Frente A
branch_labels = None
depends_on = None


# Mapping construído via inventário Task 1.
# Atualizar lista antes de rodar com base no SELECT slug FROM auth.audiences WHERE slug ~ '_'.
RENAME_MAP = [
    ("plexco_tickets", "plexco-tickets"),
    ("plexco_tasks", "plexco-tasks"),
    # adicionar outros conforme inventário
]


def upgrade() -> None:
    conn = op.get_bind()
    for old_slug, new_slug in RENAME_MAP:
        # Verifica se o old_slug existe (idempotência)
        exists = conn.execute(
            sa.text("SELECT 1 FROM auth.audiences WHERE slug = :s"), {"s": old_slug}
        ).first()
        if not exists:
            continue
        # Verifica conflito de destino
        conflict = conn.execute(
            sa.text("SELECT 1 FROM auth.audiences WHERE slug = :s"), {"s": new_slug}
        ).first()
        if conflict:
            raise RuntimeError(
                f"Cannot rename {old_slug} -> {new_slug}: destination already exists. Resolva manualmente."
            )
        conn.execute(
            sa.text(
                """
                UPDATE auth.audiences
                SET slug = :new_slug,
                    alias_slugs = COALESCE(alias_slugs, ARRAY[]::text[]) || ARRAY[:old_slug]
                WHERE slug = :old_slug
                """
            ),
            {"new_slug": new_slug, "old_slug": old_slug},
        )


def downgrade() -> None:
    conn = op.get_bind()
    for old_slug, new_slug in RENAME_MAP:
        conn.execute(
            sa.text(
                """
                UPDATE auth.audiences
                SET slug = :old_slug,
                    alias_slugs = array_remove(alias_slugs, :old_slug)
                WHERE slug = :new_slug AND :old_slug = ANY(alias_slugs)
                """
            ),
            {"old_slug": old_slug, "new_slug": new_slug},
        )
```

```python
from sqlalchemy import select, or_

async def get_audience_by_slug(db: AsyncSession, slug: str) -> Audience | None:
    stmt = select(Audience).where(
        or_(Audience.slug == slug, Audience.alias_slugs.any(slug))
    )
```

- [ ] **Step 1: Escrever helper de scan**

`plugin/percus-review/scripts/audience-kebab-grep.ps1`:

```powershell
<#
.SYNOPSIS
Detecta audiences com underscore (R7.5 violation) em arquivos do projeto.

.DESCRIPTION
Procura strings como "plexco_tickets", 'familia_x', etc. em .ts/.tsx/.py/.env*/.yaml/.yml.
Retorna lista de matches (file:line: content) ou array vazio.

.PARAMETER ScanRoot
Diretório a varrer (default: cwd)
#>
[CmdletBinding()]
param(
    [string]$ScanRoot = (Get-Location).Path
)

$pattern = '[''"](plexco|familia|painel|paid)_[a-z][a-z0-9_]*[''"]'
$extensions = @('*.ts', '*.tsx', '*.py', '*.env', '*.env.*', '*.yaml', '*.yml')

$matches = @()
foreach ($ext in $extensions) {
    $files = Get-ChildItem -Path $ScanRoot -Filter $ext -Recurse -ErrorAction SilentlyContinue |
        Where-Object { $_.FullName -notmatch '(node_modules|\.venv|__pycache__|\.git|dist|build)' }
    foreach ($f in $files) {
        Select-String -Path $f.FullName -Pattern $pattern -AllMatches | ForEach-Object {
            $matches += [pscustomobject]@{
                File    = $_.Path
                Line    = $_.LineNumber
                Match   = $_.Matches[0].Value
                Content = $_.Line.Trim()
            }
        }
    }
}

# Também busca .env sem extensão padrão
$envFiles = Get-ChildItem -Path $ScanRoot -Recurse -File -ErrorAction SilentlyContinue |
    Where-Object { $_.Name -match '^\.env' -and $_.FullName -notmatch '(node_modules|\.venv|\.git)' }
foreach ($f in $envFiles) {
    Select-String -Path $f.FullName -Pattern $pattern -AllMatches | ForEach-Object {
        $matches += [pscustomobject]@{
            File    = $_.Path
            Line    = $_.LineNumber
            Match   = $_.Matches[0].Value
            Content = $_.Line.Trim()
        }
    }
}

return $matches
```

```powershell
#requires -Version 7.0
Describe "Audience kebab enforcement (R7.5)" {
    BeforeAll {
        $script:helperPath = Join-Path $PSScriptRoot ".." "scripts" "audience-kebab-grep.ps1"
        $script:tmpRoot = Join-Path ([IO.Path]::GetTempPath()) "kebab-test-$(Get-Random)"
        New-Item -ItemType Directory -Force -Path $script:tmpRoot | Out-Null
    }

    AfterAll {
        Remove-Item -Recurse -Force $script:tmpRoot -ErrorAction SilentlyContinue
    }

    It "detecta plexco_tickets em arquivo .env" {
        $envFile = Join-Path $script:tmpRoot ".env.local"
        "PERCUS_AUTH_AUDIENCE=plexco_tickets" | Out-File $envFile -Encoding utf8

        $matches = & $script:helperPath -ScanRoot $script:tmpRoot

        $matches | Should -Not -BeNullOrEmpty
        $matches[0].Match | Should -Match 'plexco_tickets'
    }

    It "detecta audience com underscore em .ts" {
        $tsFile = Join-Path $script:tmpRoot "config.ts"
        'export const AUDIENCE = "plexco_tasks";' | Out-File $tsFile -Encoding utf8

        $matches = & $script:helperPath -ScanRoot $script:tmpRoot

        $matches.Count | Should -BeGreaterThan 0
    }

    It "NAO detecta kebab-case (canonico)" {
        # cleanup primeiro
        Get-ChildItem $script:tmpRoot -Recurse -File | Remove-Item -Force
        $envFile = Join-Path $script:tmpRoot ".env.local"
        "PERCUS_AUTH_AUDIENCE=plexco-tickets" | Out-File $envFile -Encoding utf8

        $matches = & $script:helperPath -ScanRoot $script:tmpRoot
        $matches | Should -BeNullOrEmpty
    }

    It "ignora node_modules e .venv" {
        Get-ChildItem $script:tmpRoot -Recurse -File | Remove-Item -Force
        $nodeMod = Join-Path $script:tmpRoot "node_modules/foo/.env"
        New-Item -ItemType Directory -Force -Path (Split-Path $nodeMod) | Out-Null
        "AUD=plexco_tasks" | Out-File $nodeMod -Encoding utf8

        $matches = & $script:helperPath -ScanRoot $script:tmpRoot
        $matches | Should -BeNullOrEmpty
    }

    It "self-check: canon _Novo_Projeto NAO contem audiences com underscore" {
        # Verifica que o proprio canon nao tem underscore (R7.5 self-enforcement)
        $canonRoot = (Resolve-Path (Join-Path $PSScriptRoot "../../..")).Path
        $matches = & $script:helperPath -ScanRoot $canonRoot

        # Filtrar matches que sao DOCUMENTAÇÃO de exemplo (em .md são esperados como "❌ plexco_tasks")
        $offenders = $matches | Where-Object {
            $_.File -notmatch '\.md$' -and
            $_.File -notmatch '/docs/superpowers/' -and
            $_.File -notmatch 'CANON_VERSION\.md$'
        }

        if ($offenders.Count -gt 0) {
            $offenders | ForEach-Object { Write-Host "  $($_.File):$($_.Line): $($_.Content)" }
        }
        $offenders | Should -BeNullOrEmpty -Because "R7.5: nenhuma audience com underscore em codigo do canon (md docs sao exemplos OK)"
    }
}
```
