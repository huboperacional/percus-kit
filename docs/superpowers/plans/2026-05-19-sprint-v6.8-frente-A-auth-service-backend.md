> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Adicionar schema de branding em `auth.audiences` + endpoint público `GET /tenants/by-origin` + rota admin de update branding. Foundation que destrava Frentes B/C/D/E.

**Architecture:** Alembic migration introduz 7 colunas em `auth.audiences` (branding + origins + alias_slugs) **reusando o `display_name` existente como nome do produto** (não duplica). Novo módulo `modules/tenants/` expõe rota pública com lookup `WHERE :origin = ANY(origins)`, rate-limited 100 req/min por origin. API response mapeia `display_name` → `product_name` (contrato cliente estável). Rota admin `PUT /admin/audiences/{slug}/branding` aceita `product_name` no body e grava em `display_name`. Step-up TOTP (reusa padrão existente do auth-service).

**Tech Stack:** FastAPI, SQLAlchemy 2.x async, PostgreSQL 17, Alembic, Pydantic v2, pytest-asyncio.

```python
from __future__ import annotations

from typing import Optional
from pydantic import BaseModel, ConfigDict, Field, HttpUrl


class PaletteIn(BaseModel):
    """Cores do tema do tenant. Aceita hex (#FFFFFF) ou CSS color name."""
    primary: str = Field(..., max_length=64)
    accent: str = Field(..., max_length=64)
    gradient: list[str] | None = None  # opcional, ex: ['#FFA500', '#FF6347']


class CopyIn(BaseModel):
    """Strings customizáveis. Suporta interpolação {product_name} resolvida no service."""
    login_title: str | None = None
    helper_text: str | None = None
    cta: str | None = None


class TenantConfigOut(BaseModel):
    """Response de GET /tenants/by-origin.

    `product_name` no response é mapeado de `Audience.display_name` no DB
    (NÃO existe coluna `product_name`). API/cliente vê sempre `product_name`.
    """
    audience: str  # PK column da tabela (kebab-case)
    product_name: str = Field(alias="display_name")
    logo_url: Optional[str] = None
    logo_url_dark: Optional[str] = None
    palette: Optional[PaletteIn] = None
    copy: Optional[CopyIn] = None
    support_contact_url: Optional[str] = None

    model_config = ConfigDict(from_attributes=True, populate_by_name=True)


class BrandingIn(BaseModel):
    """Body de PUT /admin/audiences/{slug}/branding."""
    product_name: str = Field(..., min_length=1, max_length=128)
    logo_url: Optional[HttpUrl] = None
    logo_url_dark: Optional[HttpUrl] = None
    palette: Optional[PaletteIn] = None
    copy: Optional[CopyIn] = None
    support_contact_url: Optional[HttpUrl] = None
```

```python
"""Lookup de tenant config por origin (Sprint v6.8 Frente A)."""
from __future__ import annotations

from sqlalchemy import select
from sqlalchemy.ext.asyncio import AsyncSession

from app.models.audience import Audience


async def get_tenant_by_origin(db: AsyncSession, origin: str) -> Audience | None:
    """Retorna o tenant cujo array `origins` contém o `origin` recebido.

    Usa Postgres ANY() operator com índice GIN em `audiences.origins`.
    """
    stmt = select(Audience).where(Audience.origins.any(origin))
    return (await db.execute(stmt)).scalar_one_or_none()
```

```python
async def check_origin_rate_limit(origin: str, *, limit: int = 100, window_sec: int = 60) -> bool:
    """Fixed-window rate limit por origin pro endpoint /tenants/by-origin.

    Origem é tratada como chave canônica diretamente (lowercase, sem trim — caller controla).
    Retorna True se request pode prosseguir, False se atingiu o limite.
    """
    key = f"auth:rl:origin:{origin.lower()}"
    # Reusa o pattern fixed-window já existente no core. Se não houver helper publico,
    # usa cliente Redis direto:
    from app.core.redis_client import get_redis
    redis = await get_redis()
    count = await redis.incr(key)
    if count == 1:
        await redis.expire(key, window_sec)
    return count <= limit
```

```python
"""Router público /tenants/by-origin (Sprint v6.8 Frente A)."""
from __future__ import annotations

from fastapi import APIRouter, Depends, HTTPException, Query, status
from sqlalchemy.ext.asyncio import AsyncSession

from app.core.db import get_db
from app.core.rate_limit import check_origin_rate_limit
from app.modules.tenants.schemas import TenantConfigOut
from app.modules.tenants.service import get_tenant_by_origin

router = APIRouter(prefix="/tenants", tags=["tenants"])


@router.get(
    "/by-origin",
    response_model=TenantConfigOut,
    responses={
        404: {"description": "Tenant não encontrado pro origin recebido."},
        429: {"description": "Rate limit excedido (100 req/min por origin)."},
    },
)
async def by_origin(
    origin: str = Query(..., description="window.location.origin do cliente"),
    db: AsyncSession = Depends(get_db),
) -> TenantConfigOut:
    """Retorna config de branding do tenant cujo origin está autorizado.

    Público (sem auth). Rate limited 100 req/min por origin.
    """
    allowed = await check_origin_rate_limit(origin)
    if not allowed:
        raise HTTPException(
            status_code=status.HTTP_429_TOO_MANY_REQUESTS,
            detail={"error": "rate_limited"},
        )

    audience = await get_tenant_by_origin(db, origin)
    if audience is None:
        raise HTTPException(
            status_code=status.HTTP_404_NOT_FOUND,
            detail={"error": "tenant_not_found"},
        )
    return TenantConfigOut.model_validate(audience)
```

```python
"""Rota admin pra update de branding por audience (Sprint v6.8 Frente A)."""
from __future__ import annotations

from fastapi import APIRouter, Depends, HTTPException, status
from sqlalchemy.ext.asyncio import AsyncSession

from app.core.db import get_db
from app.core.auth_dep import require_step_up_totp  # ajuste pro nome real
from app.models.audience import Audience
from app.modules.tenants.schemas import BrandingIn, TenantConfigOut

router = APIRouter(
    prefix="/admin/audiences",
    tags=["admin", "tenants"],
    dependencies=[Depends(require_step_up_totp)],
)


@router.put("/{slug}/branding", response_model=TenantConfigOut)
async def update_branding(
    slug: str,
    body: BrandingIn,
    db: AsyncSession = Depends(get_db),
) -> TenantConfigOut:
    """Atualiza branding/origins de uma audience. Exige step-up TOTP.

    Note: criação de audience continua em /admin/audiences/ (rota existente).
    """
    from sqlalchemy import select
    result = await db.execute(select(Audience).where(Audience.audience == slug))
    audience = result.scalar_one_or_none()
    if audience is None:
        raise HTTPException(status_code=status.HTTP_404_NOT_FOUND, detail="audience not found")

    # Aplicar campos — product_name no body grava em display_name (canonical column)
    audience.display_name = body.product_name
    audience.logo_url = str(body.logo_url) if body.logo_url else None
    audience.logo_url_dark = str(body.logo_url_dark) if body.logo_url_dark else None
    audience.palette = body.palette.model_dump() if body.palette else None
    audience.copy = body.copy.model_dump() if body.copy else None
    audience.support_contact_url = str(body.support_contact_url) if body.support_contact_url else None
    audience.origins = body.origins

    await db.commit()
    await db.refresh(audience)
    return TenantConfigOut.model_validate(audience)
```
